1   package edu.brown.cs.student.termProject;
2   
3   /**
4    * The attraction node interface is implemented by each attraction and is used for Dijkstra's.
5    */
6   public interface AttractionNode {
7   
8     /**
9      * Gets the unique Id of the node.
10     * @return Id of the node
11     */
12    String getId();
13  
14    /**
15     * Gets the non-unique name of the node.
16     * @return name of the node
17     */
18    String getName();
19  
20    /**
21     * Gets the location of the node.
22     * @return [address, city, state, postal_code]
23     */
24    String[] getLocation();
25  
26    /**
27     * Gets the latitude/longitude location of the node.
28     * @return [lat,long]
29     */
30    double[] getCoordinates();
31  
32    /**
33     * Gets price associated with a node.
34     * @return price where 1 = $, 2 = $$, 3 = $$$
35     */
36    double getPrice();
37  
38    /**
39     * Gets rating of a node.
40     * @return rating between 0 and 5
41     */
42    double getRating();
43  
44    /**
45     * Node's value heuristic.
46     * @param preferredPrice value indicating user's preferred price point
47     * @param preferredStop value indicating how much user prefers that type of stop
48     * @param dist from current node to prev
49     * @return value heuristic used in Dijkstra's PQ
50     */
51    double generateValue(double preferredPrice, double preferredStop, double dist);
52  
53    /**
54     * Sets the "cost" associated with the node to be used in Dijkstra's.
55     * @param c the cost
56     */
57    void setCost(double c);
58  
59    /**
60     * Retrieves the "cost" associated with the node to be used in Dijkstra's.
61     * @return the cost
62     */
63    double getCost();
64  
65    /**
66     * Set's distance.
67     * @param c double of distance to be set.
68     */
69    void setDistance(double c);
70  
71    /**
72     * Returns distance using node's distance calculation.
73     * @return double representing distance
74     */
75    double getDistance();
76  
77    /**
78     * sets boolean representing if node has been visited.
79     * @param c boolean
80     */
81    void setVisited(boolean c);
82  
83    /**
84     * Returned boolean of visited.
85     * @return true if visited, false otherwise.
86     */
87    boolean getVisited();
88  
89    /**
90     * Sets the number of previous nodes in route.
91     * @param c int of num previous nodes
92     */
93    void setNumPrev(int c);
94  
95    /**
96     * Get's number of previous stops.
97     * will be used in value heuristic
98     * @return integer of number of prev
99     */
100   int getNumPrev();
101 
102   /**
103    * integer associated with Attraction type.
104    * 0 = Museum
105    * 1 = Park
106    * 2 = Restaurant
107    * 3 = Shop
108    * @return integer
109    */
110   int getType();
111   /**
112    * returns the node's "value".
113    * @return the value
114    */
115   double getValue();
116 
117   /**
118    * returns the node's number of reviews.
119    * @return the number of revies
120    */
121   double getNumReviews();
122 
123 }
