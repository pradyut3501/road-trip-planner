106   /**
107    * runs the spark server.
108    *
109    * @param port given number
110    */
111   private void runSparkServer(int port) {
112     Spark.port(port);
113     Spark.externalStaticFileLocation("src/main/resources/static");
114     Spark.exception(Exception.class, new ExceptionPrinter());
115 
116 
117     Spark.options("/*", (request, response) -> {
118       String accessControlRequestHeaders = request.headers("Access-Control-Request-Headers");
119       if (accessControlRequestHeaders != null) {
120         response.header("Access-Control-Allow-Headers", accessControlRequestHeaders);
121       }
122 
123       String accessControlRequestMethod = request.headers("Access-Control-Request-Method");
124 
125       if (accessControlRequestMethod != null) {
126         response.header("Access-Control-Allow-Methods", accessControlRequestMethod);
127       }
128 
129       return "OK";
130     });
131 
132     FreeMarkerEngine freeMarker = createEngine();
133 
134     // Setup Spark Routes
135     Spark.before((request, response) -> response.header("Access-Control-Allow-Origin", "*"));
136     Spark.exception(Exception.class, new ExceptionPrinter());
137     Spark.post("/route", new RouteHandler());
138   }

152   /**
153    * Class that handles getting nearest traversable node for maps3 frontend.
154    */
155   private static class RouteHandler implements Route {
156     @Override
157     public Object handle(Request request, Response response) throws Exception {
158       String error = "";
159       List<AttractionNode> route = new ArrayList<>();
160       try {
161         JSONObject data = new JSONObject(request.body());
162         //preferences which will be used for value heuristic
163         int costPreference = data.getInt("costPref");
164         double restaurantValue = data.getDouble("restValue");
165         double museumValue = data.getDouble("musValue");
166         double parkValue = data.getDouble("parkValue");
167         double shopValue = data.getDouble("shopValue");
168         double[] preferredStop = {museumValue, parkValue, restaurantValue, shopValue};
169         double originLat = data.getDouble("originLat");
170         double originLon = data.getDouble("originLon");
171         double destLat = data.getDouble("destLat");
172         double destLon = data.getDouble("destLon");
173         double maxDist = data.getDouble("maxDist");
174         double middleLat = 0.0;
175         double middleLong = 0.0;
176         try {
177           middleLat = data.getDouble("middleLat");
178           middleLong = data.getDouble("middleLon");
179         } catch (JSONException e) {
180           System.out.println("Problem getting middle lat: " + e);
181         }
182         System.out.println("Middle Lat " + middleLat + " and middle long " + middleLong);
183         int numStops = 0;
184         try {
185           numStops = data.getInt("stopPref");
186         } catch (Exception e) {
187           error = "Number of stops input is in the wrong format. Must be an integer greater than"
188               + " 0";
189           return new Gson().toJson(ImmutableMap.of("route", route, "error_message", error));
190         }
191         List<String> categories = new ArrayList<>();
192         if (restaurantValue >= 0.3) {
193           categories.add("Restaurant");
194         }
195         if (parkValue >= 0.3) {
196           categories.add("Park");
197         }
198         if (museumValue >= 0.3) {
199           categories.add("Museum");
200         }
201         if (shopValue >= 0.3) {
202           categories.add("Shop");
203         }
204         Dijkstra>Dijkstra dijkstra = new Dijkstra(APICONNECTION);
205         if (numStops > 0) {
206           double miles = dijkstra.distanceFormulaAPI(originLat,
207               originLon, destLat, destLon) / 1000 / Constants.MILES_TO_KILOMETERS;
208           System.out.println("Google Maps API: " + miles);
209           //check to make sure the inputted max distance is not greater than the minimum distance
210           if ((maxDist * Constants.MILES_TO_KILOMETERS) < (dijkstra.distanceFormulaAPI(originLat,
211               originLon, destLat, destLon) / 1000)) {
212             error = "It is impossible to complete the trip in " + maxDist + " miles";
213             System.out.println(error);
214             return new Gson().toJson(ImmutableMap.of("route", route, "error_message", error));
215           }
216           //if the user has to make an intermediate step
217           if (middleLat != 0.0 && middleLong != 0.0) {
218             System.out.println("Intermediate Stop!");
219             double proportionBeforeStop = dijkstra.distanceFormulaAPI(originLat, originLon,
220                 middleLat,
221                 middleLong) / (dijkstra.distanceFormulaAPI(originLat, originLon, middleLat,
222                 middleLong) + dijkstra.distanceFormulaAPI(middleLat, middleLong, destLat, destLon));
223             double firstHalf = proportionBeforeStop * numStops;
224             int numStopsFirstHalf = (int) Math.floor(firstHalf);
225             int numStopsSecondHalf = numStops - numStopsFirstHalf;
226             System.out.println("First half is : " + numStopsFirstHalf + " and second half is "
227                 + numStopsSecondHalf);
228 
229             dijkstra
230                 .setPreferences(preferredStop, costPreference, BoundingBox.findAttractionsBetween(
231                     new double[] {originLat, originLon},
232                     new double[] {middleLat, middleLong}, categories, numStopsFirstHalf,
233                     costPreference));
234             List<AttractionNode> route1 = dijkstra.execute(new double[] {originLat, originLon},
235                 new double[] {middleLat, middleLong}, numStopsFirstHalf);
236             //set new preferences
237             dijkstra
238                 .setPreferences(preferredStop, costPreference, BoundingBox.findAttractionsBetween(
239                     new double[] {middleLat, middleLong},
240                     new double[] {destLat, destLon}, categories, numStopsSecondHalf,
241                     costPreference));
242             List<AttractionNode> route2 = dijkstra.execute(new double[] {middleLat, middleLong},
243                 new double[] {destLat, destLon}, numStopsSecondHalf);
244 
245             route.addAll(route1);
246             //dumby "node" to represent the user inputted intermediate stop
247             route.add(new Park("0", "Intermediate Stop", new String[] {""},
248                 new double[] {middleLat, middleLong}, 0.0, 0.0, 0.0));
249             route.addAll(route2);
250             System.out.println("The length of your first part is " + route1.size() + " and your "
251                 + "second part is " + route2.size());
252 
253           } else {
254             List<AttractionNode> attractions = BoundingBox.findAttractionsBetween(
255                 new double[] {originLat, originLon},
256                 new double[] {destLat, destLon}, categories, numStops, costPreference);
257             System.out.println("number of nodes: " + attractions.size());
258             dijkstra.setPreferences(preferredStop, costPreference, attractions);
259             route = dijkstra.execute(new double[] {originLat, originLon},
260                 new double[] {destLat, destLon}, numStops);
261           }
262         } else if (numStops == 0) {
263           error = "This route has no stops";
264           return new Gson().toJson(ImmutableMap.of("route", route, "error_message", error));
265         } else if (numStops < 0) {
266           error = "The number of stops cannot be negative";
267           return new Gson().toJson(ImmutableMap.of("route", route, "error_message", error));
268         }
269         double routeLength = 0;
270         for (int i = 1; i < route.size(); i++) {
271           routeLength += dijkstra.distanceFormulaAPI(route.get(i - 1).getCoordinates()[0],
272               route.get(i - 1).getCoordinates()[1], route.get(i).getCoordinates()[0],
273               route.get(i).getCoordinates()[1]);
274         }
275         routeLength += dijkstra.distanceFormulaAPI(route.get(0).getCoordinates()[0],
276             route.get(0).getCoordinates()[1], originLat, originLon);
277         routeLength += dijkstra.distanceFormulaAPI(route.get(route.size() - 1).getCoordinates()[0],
278             route.get(route.size() - 1).getCoordinates()[1], destLat, destLon);
279         routeLength = (routeLength / 1000) / Constants.MILES_TO_KILOMETERS;
280         System.out.println("Dijkstra's Route is this length " + route.size() + " and this many "
281             + "miles: " + routeLength);
282         if (routeLength > maxDist) {
283           double diff = Math.ceil(routeLength - maxDist);
284           error = "Unfortunately, we cannot plan you a roadtrip under " + maxDist + " miles. You"
285               + " would need to increase your distance constraint by " + diff;
286           System.out.println(error);
287           route = new ArrayList<>();
288         }
289         for (AttractionNode r : route) {
290           System.out.println(r.getName() + " "
291               + " and a cost of " + r.getCost() + " and a value of " + r.getValue());
292         }
293 
294       } catch (Exception e) {
295         System.out.println("problem with dijkstras");
296         e.printStackTrace();
297       }
298 
299       Map<String, Object> variables = ImmutableMap.of("route", route, "error_message", error);
300 //      try {
301 //      } catch (Exception e) {
302 //        System.out.println("something went wrong here");
303 //        e.printStackTrace();
304 //      }
305       return new Gson().toJson(variables);
306     }
307   }

312   private static class ExceptionPrinter implements ExceptionHandler {
313     @Override
314     public void handle(Exception e, Request req, Response res) {
315       res.status(500);
316       StringWriter stacktrace = new StringWriter();
317       try (PrintWriter pw = new PrintWriter(stacktrace)) {
318         pw.println("<pre>");
319         e.printStackTrace(pw);
320         pw.println("</pre>");
321       }
322       res.body(stacktrace.toString());
323     }
