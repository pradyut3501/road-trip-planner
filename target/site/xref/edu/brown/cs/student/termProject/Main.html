117   /**
118    * runs the spark server.
119    * @param port given number
120    */
121   private void runSparkServer(int port) {
122     Spark.port(port);
123     Spark.externalStaticFileLocation("src/main/resources/static");
124     Spark.exception(Exception.class, new ExceptionPrinter());
125 
126 
127     Spark.options("/*", (request, response) -> {
128       String accessControlRequestHeaders = request.headers("Access-Control-Request-Headers");
129       if (accessControlRequestHeaders != null) {
130         response.header("Access-Control-Allow-Headers", accessControlRequestHeaders);
131       }
132 
133       String accessControlRequestMethod = request.headers("Access-Control-Request-Method");
134 
135       if (accessControlRequestMethod != null) {
136         response.header("Access-Control-Allow-Methods", accessControlRequestMethod);
137       }
138 
139       return "OK";
140     });
141 
142     FreeMarkerEngine freeMarker = createEngine();
143 
144     // Setup Spark Routes
145     Spark.before((request, response) -> response.header("Access-Control-Allow-Origin", "*"));
146     Spark.exception(Exception.class, new ExceptionPrinter());
147     Spark.post("/route", new RouteHandler());
148   }

256   /**
257    * Display an error page when an exception occurs in the server.
258    */
259   private static class ExceptionPrinter implements ExceptionHandler {
260     @Override
261     public void handle(Exception e, Request req, Response res) {
262       res.status(500);
263       StringWriter stacktrace = new StringWriter();
264       try (PrintWriter pw = new PrintWriter(stacktrace)) {
265         pw.println("<pre>");
266         e.printStackTrace(pw);
267         pw.println("</pre>");
268       }
269       res.body(stacktrace.toString());
270     }
