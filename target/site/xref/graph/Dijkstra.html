36    public List<AttractionNode> execute(double[] starting, double[] ending, int numStops){
37      double pathDistance = distanceFormula(starting[0], starting[1], ending[0], ending[1]);
38      visited = new HashMap<>();
39      distances = new HashMap<>();
40      previous = new HashMap<>();
41      PriorityQueue<AttractionNode> pq = new PriorityQueue(new Comparator<AttractionNode>() {
42        public int compare(AttractionNode/edu/brown/cs/student/termProject/AttractionNode.html#AttractionNode">AttractionNode o1, AttractionNode o2) {
43          if ((o1.getCost() + o1.generateValue(costPreference, preferredStop[o1.getType()])) >
44            (o2.getCost()) + o2.generateValue(costPreference, preferredStop[o2.getType()])) {
45            return 1;
46          }
47          if ((o2.getCost() + o2.generateValue(costPreference, preferredStop[o2.getType()])) >
48            (o1.getCost() + o1.generateValue(costPreference, preferredStop[o1.getType()]))) {
49            return -1;
50          }
51          return 0;
52        }
53      });
54      List <AttractionNode> shortestPath = new ArrayList<>();
55      double[] target = ending; //get the target
56      if (starting == ending) { //stop if the start and end node are the same
57        return null;
58      }
59      //Make a dumby attraction node to represent start and end? may be a better way to handle this
60      AttractionNode start = new Park("0", "starting Node", new String[0], starting,0.0,0.0 );
61      nodes.add(start);
62      AttractionNode end = new Park("0", "ending Node", new String[0], target,0.0,0.0 );
63      nodes.add(end);
64      distances.replace(end, 0.0); //setting the distance of end node to 0
65      pq.add(start);
66      double minLon = 100000;
67      double maxLon = -100000;
68      for (AttractionNode node: nodes){
69       // System.out.println(node.getLocation()[2]);
70        visited.put(node, false);
71        previous.put(node, null);
72        node.setCost(Double.POSITIVE_INFINITY);
73      }
74      start.setCost(0.0);
75      System.out.println("Ideal Spacing is" + pathDistance/numStops);
76  
77      while (!(pq.isEmpty()) && !(visited.get(end))) {
78        AttractionNode current = pq.poll();
79        visited.replace(current,true); //mark the popped node as visited
80        List<AttractionNode> connectedNodes = getConnectedNodes(current, end, pathDistance, numStops);
81          for (AttractionNode node: connectedNodes) {
82            //if the current nodes cost plus the distance between the current and the node you are
83            // examining is less than the node you are examining's cost
84            double edgeWeight = distanceFormula(current.getCoordinates()[0],
85              current.getCoordinates()[1], node.getCoordinates()[0], node.getCoordinates()[1]);
86            if ((current.getCost() + edgeWeight) < node.getCost()) {
87              node.setCost(current.getCost() + edgeWeight); //Will need to add value to this!!
88              previous.replace(node, current);
89              pq.add(node);
90            }
91          }
92          }
93      AttractionNode curr = end;
94     // System.out.println("Previous node to the end: " + previous.get(end));
95      //gets the shortest path by looking at previous
96      //start with the end, while the previous node is not the start add to current path
97      if(previous.get(curr)== start){
98        System.out.println("shortest path is direct one :(");
99      }
100     while(previous.get(curr)!= start){
101       shortestPath.add(previous.get(curr));
102       curr = previous.get(curr);
103       //System.out.println(curr);
104     }
105    // shortestPath.add(start);
106     Collections.reverse(shortestPath);
107     return shortestPath;
108   }

142   private List<AttractionNode> getConnectedNodes(AttractionNode node,
143                                                  AttractionNode end, double distance,int numStops){
144     double spacing = distance/numStops;
145     List<AttractionNode> connects = nodes;
146     List<AttractionNode> toRemove = new ArrayList<>(); //To avoid concurrent modification error
147     for(AttractionNode n: connects){
148       if (distanceFormula(n.getCoordinates()[0],
149         n.getCoordinates()[1], end.getCoordinates()[0], end.getCoordinates()[1]) >=
150         distanceFormula(node.getCoordinates()[0], node.getCoordinates()[1],end.getCoordinates()[0],
151           end.getCoordinates()[1])
152       ){
153         toRemove.add(n); //remove the node from the list of possible connecting nodes if it is
154         // farther from the target than the current node
155       }
156     }
157     for(AttractionNode r: toRemove){
158       connects.remove(r);
159     }
160 //    System.out.println("The node " + node.getName() + "has this many connections: " + connects
161 //    .size());
162     Collections.sort(connects, new Comparator<AttractionNode>(){
163       @Override
164       //this comparator will sort the nodes based on HOW CLOSE THEY are to the target spacing,
165       // i.e. nodes that are spaced closest to the "spacing" metric will be considered connected
166       public int compare(AttractionNode/edu/brown/cs/student/termProject/AttractionNode.html#AttractionNode">AttractionNode o1, AttractionNode o2) {
167         double distance1 = distanceFormula(o1.getCoordinates()[0], o1.getCoordinates()[1],
168           node.getCoordinates()[0], node.getCoordinates()[1]);
169         double distance2 = distanceFormula(o2.getCoordinates()[0], o2.getCoordinates()[1],
170           node.getCoordinates()[0], node.getCoordinates()[1]);
171         if (Math.abs(distance1-spacing) > Math.abs(distance2-spacing)){
172           return 1;
173         }
174         if (Math.abs(distance2-spacing) > Math.abs(distance1-spacing)){
175           return -1;
176         }
177         else{
178           return 0;
179         }
180       }
181     });
182     if (node == end){
183       System.out.println("The connected nodes to the end re" + connects);
184     }
185     if(connects.size()>=Constants.NUM_CONNECTIONS){
186       return connects.subList(0,Constants.NUM_CONNECTIONS);
187     }
188    else if(connects.size() >= 1){
189      return connects;
190     }
191    else{
192      return new ArrayList<>();
193     }
194   }
