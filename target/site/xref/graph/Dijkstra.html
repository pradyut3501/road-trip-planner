39    public List<AttractionNode> execute(double[] starting, double[] ending, int numStops) {
40      double pathDistance = distanceFormula(starting[0], starting[1], ending[0], ending[1]);
41      visited = new HashMap<>();
42      distances = new HashMap<>();
43      previous = new HashMap<>();
44      PriorityQueue<AttractionNode> pq = new PriorityQueue(new Comparator<AttractionNode>() {
45        public int compare(AttractionNode/edu/brown/cs/student/termProject/AttractionNode.html#AttractionNode">AttractionNode o1, AttractionNode o2) {
46          if ((o1.getCost() + o1.generateValue(costPreference, preferredStop[o1.getType()], pathDistance)) >
47            (o2.getCost()) + o2.generateValue(costPreference, preferredStop[o2.getType()],pathDistance)) {
48            return 1;
49          }
50          if ((o2.getCost() + o2.generateValue(costPreference, preferredStop[o2.getType()],pathDistance)) >
51            (o1.getCost() + o1.generateValue(costPreference, preferredStop[o1.getType()],pathDistance))) {
52            return -1;
53          }
54          return 0;
55        }
56      });
57      List <AttractionNode> shortestPath = new ArrayList<>();
58      double[] target = ending; //get the target
59      if (starting == ending) { //stop if the start and end node are the same
60        return null;
61      }
62      //Make a dumby attraction node to represent start and end? may be a better way to handle this
63      AttractionNode start = new Park("0", "starting Node", new String[0], starting,
64        0.0,0.0, 0.0 );
65      nodes.add(start);
66      AttractionNode end = new Park("0", "ending Node", new String[0], target,
67        0.0,0.0, 0.0 );
68      nodes.add(end);
69      distances.replace(end, 0.0); //setting the distance of end node to 0
70      pq.add(start);
71      for (AttractionNode node: nodes){
72        visited.put(node, false);
73        previous.put(node, null);
74        node.setCost(Double.POSITIVE_INFINITY);
75      }
76      start.setCost(0.0);
77      System.out.println("Ideal Spacing is" + pathDistance/numStops);
78      while (!(pq.isEmpty()) && !(visited.get(end))) {
79        AttractionNode current = pq.poll();
80        visited.replace(current,true); //mark the popped node as visited
81        List<AttractionNode> connectedNodes = getConnectedNodes(current, end, pathDistance, numStops);
82          for (AttractionNode node: connectedNodes) {
83            //if the current nodes cost plus the distance between the current and the node you are
84            // examining is less than the node you are examining's cost
85            double edgeWeight = distanceFormula(current.getCoordinates()[0],
86              current.getCoordinates()[1], node.getCoordinates()[0], node.getCoordinates()[1]);
87            if ((current.getCost() + edgeWeight) < node.getCost()) {
88              node.setCost(current.getCost() + edgeWeight); //reset the cost
89              previous.replace(node, current); //reset the previous pointer
90              pq.add(node);
91            }
92          }
93          }
94      AttractionNode curr = end;
95      if(previous.get(curr)== start){
96        System.out.println("shortest path is direct one :(");
97      }
98      while(previous.get(curr)!= start){
99        shortestPath.add(previous.get(curr));
100       curr = previous.get(curr);
101     }
102     Collections.reverse(shortestPath); //want path to be in order from start to end
103     nodes.remove(start);
104     nodes.remove(end); //remove the dumby nodes
105     return shortestPath;
106   }

126       if (response.routes.length > 0) {
127         double dist = 0;
128         for (int i =0; i < response.routes[0].legs.length; i++) {
129           dist += response.routes[0].legs[i].distance.inMeters;
130         }
131         return dist;
132       }
133     } catch (ApiException | InterruptedException | IOException e) {
134       e.printStackTrace();
135     }
136     return 0.0;

177   /**
178    * Gets a list of nodes "connected" to the current node as specified by the "ideal spacing"
179    * between nodes. Works to evenly space out stops on the roadtrip
180    * @param node the node to find connections from
181    * @param end the end node
182    * @param distance the distance between the start and end nodes
183    * @param numStops the number of stops ideally between start and end node
184    * @return list of connected nodes
185    */
186   private List<AttractionNode> getConnectedNodes(AttractionNode node,
187                                                  AttractionNode end, double distance,int numStops){
188     double spacing = distance/numStops;
189     List<AttractionNode> connects = nodes;
190     List<AttractionNode> toRemove = new ArrayList<>(); //To avoid concurrent modification error
191     for(AttractionNode n: connects){
192       if (distanceFormula(n.getCoordinates()[0],
193         n.getCoordinates()[1], end.getCoordinates()[0], end.getCoordinates()[1]) >=
194         distanceFormula(node.getCoordinates()[0], node.getCoordinates()[1],end.getCoordinates()[0],
195           end.getCoordinates()[1])
196       ){
197         toRemove.add(n); //remove the node from the list of possible connecting nodes if it is
198         // farther from the target than the current node
199       }
200     }
201     for(AttractionNode r: toRemove){
202       connects.remove(r);
203     }
204     Collections.sort(connects, new Comparator<AttractionNode>(){
205       @Override
206       //this comparator will sort the nodes based on HOW CLOSE THEY are to the target spacing,
207       // i.e. nodes that are spaced closest to the "spacing" metric will be considered connected
208       public int compare(AttractionNode/edu/brown/cs/student/termProject/AttractionNode.html#AttractionNode">AttractionNode o1, AttractionNode o2) {
209         double distance1 = distanceFormula(o1.getCoordinates()[0], o1.getCoordinates()[1],
210           node.getCoordinates()[0], node.getCoordinates()[1]);
211         double distance2 = distanceFormula(o2.getCoordinates()[0], o2.getCoordinates()[1],
212           node.getCoordinates()[0], node.getCoordinates()[1]);
213         if (Math.abs(distance1-spacing) > Math.abs(distance2-spacing)) {
214           return 1;
215         }
216         if (Math.abs(distance2-spacing) > Math.abs(distance1-spacing)) {
217           return -1;
218         }
219         else{
220           return 0;
221         }
222       }
223     });
224     if (node == end){
225       System.out.println("The connected nodes to the end re" + connects);
226     }
227     //return a list of the number of connections specified in the constants class. Pick the n
228     // closest elements to the "ideal distance"
229     if(connects.size()>=Constants.NUM_CONNECTIONS) {
230       return connects.subList(0,Constants.NUM_CONNECTIONS);
231     }
232     //if the list of potential connections is less than the specified number of connections for
233     // each node just return the whole list
234    else if(connects.size() >= 1) {
235      return connects;
236     }
237    else{
238      return new ArrayList<>();
239     }
240   }
