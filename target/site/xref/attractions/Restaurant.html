17    private double value;

21    private int numPrev = 0;
22    private String nodeType = "restaurant";
23    private double numReviews;
24  
25    /**
26     * The constructor sets the fields
27     * @param RestaurantId the id
28     * @param RestaurantName the name of the stop
29     * @param loc the address of the Restaurant in an array
30     * @param coords the latitude/longitude coordinates
31     * @param p the price associated with the stop
32     * @param rate the five star rating
33     * @param reviewCount number of reviews left at this establishment
34     */
35    public Restaurant(String RestaurantId, String RestaurantName, String[] loc, double[] coords,
36                     Double p, Double rate,  double reviewCount){
37      id = RestaurantId;
38      name = RestaurantName;
39      location = loc;
40      coordinates = coords;
41      price = p;
42      rating = rate;
43      cost = Double.POSITIVE_INFINITY;
44      value = 0;
45      numReviews = reviewCount;
46  
47    }

79    @Override
80    public double generateValue(double PreferredPrice, double PreferredStop, double distance) {
81      double restaurantValue = PreferredStop;
82      value = (1- restaurantValue/Constants.VALUE_BOUND) * distance;
83      value = value + (1- numReviews/Constants.AVERAGE_REVIEWS_RESTAURANTS) * distance;
84     //value = (Constants.VALUE_BOUND- restaurantValue) * Constants.VALUE_SCALE_RESTAURANTS;
85      //value = value * (1 - rating *.1);
86      value = value + (1 - rating/Constants.MAX_RATING) * distance;
87      value = value + (Math.abs(price-PreferredPrice)) * distance;
88      value = value * Constants.VALUE_SCALE;
89    //  System.out.println("restaurant value is: " + value);
90      return value;
91    }
