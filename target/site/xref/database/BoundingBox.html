16  /**
17   * Class with static functions to find all the attractions within a bounding box
18   * around two sets of coordinates.
19   */
20  public class BoundingBox {
21  
22    /**
23     * Creates a bounding box around the given starting and end positions,
24     * and returns a list of the AttractionNodes from the yelp-database within
25     * this box that are of the given categories.
26     *
27     * @param coords1 - [lat, lon] of the start position
28     * @param coords2 - [lat, lon] of the end position
29     * @param categories - list of acceptable attraction categories
30     * @return list of attraction nodes of the given categories within a
31     * box around the given positions
32     */
33    public static List<AttractionNode> findAttractionsBetween(
34            double[] coords1, double[] coords2, List<String> categories) {
35  
36      double[] boundingBoxBounds = findBoundingBoxBounds(coords1, coords2);
37  
38      double[] expandedBoundingBoxBounds = expandBoundingBoxBounds(boundingBoxBounds, 2.0);
39  
40      try {
41        return findAttractionsWithinBoundingBox(expandedBoundingBoxBounds, categories);
42      } catch (SQLException e) {
43        throw new IllegalArgumentException("ERROR: Error while connecting to SQL database");
44      }
45    }
46  
47    /**
48     * Finds the bounds for a bounding box to be created around the given coordinates.
49     *
50     * @param coords1 - [lat, lon] of the first position
51     * @param coords2 - [lat, lon] of the second position
52     * @return [start lat, end lat, start lon, end lon] of the bounding box
53     */
54    public static double[] findBoundingBoxBounds(double[] coords1, double[] coords2) {
55      double[] bounds = new double[4];
56  
57      if (coords1[0] < coords2[0]) {
58        bounds[0] = coords1[0];
59        bounds[1] = coords2[0];
60      } else {
61        bounds[0] = coords2[0];
62        bounds[1] = coords1[0];
63      }
64  
65      if (coords1[1] < coords2[1]) {
66        bounds[2] = coords1[1];
67        bounds[3] = coords2[1];
68      } else {
69        bounds[2] = coords2[1];
70        bounds[3] = coords1[1];
71      }
72  
73      return bounds;
74    }
75  
76    /**
77     * Extends the given bounding box expansionFactor units in every direction.
78     *
79     * @param boundingBoxBounds - original bounding box to be expanded
80     * @param expansionFactor - number of units to expand the bounding box by in every direction
81     * @return [start lat, end lat, start lon, end lon] of the expanded bounding box
82     */
83    public static double[] expandBoundingBoxBounds(double[] boundingBoxBounds, double expansionFactor) {
84      double[] expandedBoundingBoxBounds = new double[4];
85  
86      expandedBoundingBoxBounds[0] = boundingBoxBounds[0] - expansionFactor;
87      expandedBoundingBoxBounds[1] = boundingBoxBounds[1] + expansionFactor;
88      expandedBoundingBoxBounds[2] = boundingBoxBounds[2] - expansionFactor;
89      expandedBoundingBoxBounds[3] = boundingBoxBounds[3] + expansionFactor;
90  
91      return expandedBoundingBoxBounds;
92    }

94    /**
95     * Returns a list of all the attractions in the yelp database within the given
96     * bounding box that are of the given categories.
97     *
98     * @param boundingBoxBounds - [start lat, end lat, start lon, end lon] of the bounding box
99     * @param categories - list of acceptable attraction categories
100    * @return list of attraction nodes of the given categories within the given bounding box
101    * @throws SQLException - if yelp database cannot be successfully queried
102    */
103   public static List<AttractionNode> findAttractionsWithinBoundingBox(
104           double[] boundingBoxBounds, List<String> categories) throws SQLException {
105     Connection conn = Database.getYelpDatabaseConnection();
106 
107     String query = "SELECT * FROM yelp_business " +
108             "WHERE (latitude BETWEEN ? and ?) " +
109             "AND (longitude BETWEEN ? AND ?)" +
110             "AND (stars > 3.0)" +
111             "AND (review_count > 10)";
112 
113     PreparedStatement prep = conn.prepareStatement(query);
114     prep.setDouble(1, boundingBoxBounds[0]);
115     prep.setDouble(2, boundingBoxBounds[1]);
116     prep.setDouble(3, boundingBoxBounds[2]);
117     prep.setDouble(4, boundingBoxBounds[3]);
118 
119     ResultSet rs = prep.executeQuery();
120 
121     List<AttractionNode> attractionsWithinBox = new ArrayList<>();
122 
123     while (rs.next()) {
124 
125       String id = rs.getString("business_id");
126       String name = rs.getString("name");
127 
128       String address = rs.getString("address");
129       String city = rs.getString("city");
130       String state = rs.getString("state");
131       String postalCode = rs.getString("postal_code");
132       String[] location = new String[]{address, city, state, postalCode};
133 
134       double lat = rs.getDouble("latitude");
135       double lon = rs.getDouble("longitude");
136       double[] coords = new double[]{lat, lon};
137 
138       double price = 0; //TODO: Decide on what to do for price
139 //      String attributesList = rs.getString("attributes");
140 //      if (attributesList.contains("RestaurantsPriceRange2")) {
141 //        System.out.println("HERE");
142 //        System.out.println(attributesList);
143 //      }
144 
145       double rating = rs.getDouble("stars");
146 
147       String categoriesList = rs.getString("categories");
148 
149       if ((categoriesList.contains("Restaurants")
150               || categoriesList.contains("Food")
151               || categoriesList.contains("Bars"))
152               && categories.contains("Restaurant")) {
153         AttractionNode nextAttraction = new Restaurant(
154                 id, name, location, coords, price, rating);
155         attractionsWithinBox.add(nextAttraction);
156       } else if (categoriesList.contains("Museums") && categories.contains("Museum")) {
157         AttractionNode nextAttraction = new Museum(
158                 id, name, location, coords, price, rating);
159         attractionsWithinBox.add(nextAttraction);
160       } else if (categoriesList.contains("Parks") && categories.contains("Park")) {
161         AttractionNode nextAttraction = new Park(
162                 id, name, location, coords, price, rating);
163         attractionsWithinBox.add(nextAttraction);
164       } else if (categoriesList.contains("Shopping") && categories.contains("Shop")) {
165         AttractionNode nextAttraction = new Shop(
166                 id, name, location, coords, price, rating);
167         attractionsWithinBox.add(nextAttraction);
168       }
169     }
170     prep.close();
171     rs.close();
172 
173     return attractionsWithinBox;
174   }
