44    public static List<AttractionNode> findAttractionsBetween(
45  
46            double[] coords1, double[] coords2, List<String> categories, int prefNumStops, int costPref) {
47  
48      double[] boundingBoxBounds = findBoundingBoxBounds(coords1, coords2);

50      double[] expandedBoundingBoxBounds = expandBoundingBoxBounds(boundingBoxBounds, 2.0);

55        throw new IllegalArgumentException("ERROR: Error while connecting to SQL database");

59    /**
60     * Finds the bounds for a bounding box to be created around the given coordinates.
61     *
62     * @param coords1 - [lat, lon] of the first position
63     * @param coords2 - [lat, lon] of the second position
64     * @return [start lat, end lat, start lon, end lon] of the bounding box
65     */
66    public static double[] findBoundingBoxBounds(double[] coords1, double[] coords2) {
67      double[] bounds = new double[4];
68  
69      if (coords1[0] < coords2[0]) {
70        bounds[0] = coords1[0];
71        bounds[1] = coords2[0];
72      } else {
73        bounds[0] = coords2[0];
74        bounds[1] = coords1[0];
75      }
76  
77      if (coords1[1] < coords2[1]) {
78        bounds[2] = coords1[1];
79        bounds[3] = coords2[1];
80      } else {
81        bounds[2] = coords2[1];
82        bounds[3] = coords1[1];
83      }
84  
85      return bounds;
86    }

88    /**
89     * Extends the given bounding box expansionFactor units in every direction.
90     *
91     * @param boundingBoxBounds - original bounding box to be expanded
92     * @param expansionFactor - number of units to expand the bounding box by in every direction
93     * @return [start lat, end lat, start lon, end lon] of the expanded bounding box
94     */
95    public static double[] expandBoundingBoxBounds(double[] boundingBoxBounds, double expansionFactor) {
96      double[] expandedBoundingBoxBounds = new double[4];
97  
98      expandedBoundingBoxBounds[0] = boundingBoxBounds[0] - expansionFactor;
99      expandedBoundingBoxBounds[1] = boundingBoxBounds[1] + expansionFactor;
100     expandedBoundingBoxBounds[2] = boundingBoxBounds[2] - expansionFactor;
101     expandedBoundingBoxBounds[3] = boundingBoxBounds[3] + expansionFactor;
102 
103     return expandedBoundingBoxBounds;
104   }
